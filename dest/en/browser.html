AJAX (GET)

        <html>
        <script src="lightHttp.min.js">
        <script>
        var http = new window.lightHttp();
        http.ajax("test.html", {"count": 20}, function (content) {
            console.log(content);
        });
        </script>
        </html>

AJAX (GET) Promise

        <html>
        <script src="lightHttp.min.js">
        <script>
        var http = new window.lightHttp();
        http.ajax("test.html", {"count": 20})
            .then(function (content) {
                console.log(content);
            });
        </script>
